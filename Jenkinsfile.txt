pipeline {
    agent any

    environment {
        DOCKER_HUB_CREDENTIALS = credentials('Dockerhub_pwd')
    }

    stages {
        stage('GIT') {
            steps {
                git branch: 'main', 
                    url: 'https://github.com/ImemAyachi/devops'
            }
        }

        stage('MAVEN') {
            steps {
                sh "mvn -version"
            }
        }

        stage('Nettoyage et construction') {
            steps {
                sh "mvn clean install -DskipTests"
            }
        }

        stage('Docker Build') {
            steps {
                sh "docker build -t imem11/test1:v1 ."
            }
        }

        stage('Docker Push') {
            steps {
              
                sh "echo $DOCKER_HUB_CREDENTIALS_PSW | docker login -u $DOCKER_HUB_CREDENTIALS_USR --password-stdin"
                
              
                sh "docker push imem11/test1:v1"
            }
        }
    }
}
//dockerfile
FROM eclipse-temurin:17-jdk-jammy

WORKDIR /app
COPY target/*.jar app.jar

EXPOSE 8080

ENTRYPOINT ["java", "-jar", "app.jar"]
